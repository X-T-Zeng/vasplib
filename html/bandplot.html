function [fig,ax]=bandplot(EIGENCAR,Ecut,titlestring,color,klist_l,kpoints_l,kpoints_name,fontname,fig,ax)
%--------  init  --------
    import +vasplib_tool.*
%--------  narg  --------
    if nargin < 9
       [fig,ax] = creat_figure();
    end
    if nargin < 8
        fontname="Helvetica";
    end
    if nargin < 7
    end
    if nargin < 6
    end
    if nargin < 5
        [~,klist_l,~,kpoints_l,kpoints_name]=kpathgen3D();

    end
    if nargin < 4
        color = [rand,rand,rand];
    end
    if nargin < 3
        dirname = pwd;
        dirname = strsplit(dirname,'/');
        titlestring = dirname(length(dirname));
        titlestring = strrep(titlestring,'_','-');
    end
    if nargin < 2
        Ecut=[-3,3];
    end
    if nargin < 1
        EIGENCAR=EIGENVAl_read();
    end
%--------  juge  --------
	if ~strcmp(class(color),'double')
        color = str2rgb(color);
    end
%--------  init  --------
    Nbands=size(EIGENCAR,1);
    klist=klist_l;
    xmin=klist_l(1);
    xmax=klist_l(length(klist_l));
    Xcut = [xmin,xmax];
    ax.XLim = Xcut;
    ax.YLim = Ecut;
%--------reference -------
% reference line
    X=[xmin xmax];
    Y=[Ecut(1) Ecut(2)];
    line(X,[0 0],'LineWidth',0.1,'Color',[0. 0. 0.],'DisplayName','fermi-level')
    for i=1:(length(kpoints_name)-2)
        X=[kpoints_l(i+1) kpoints_l(i+1)];
        line(X,Y,'LineWidth',0.1,'Color',[0 0. 0.],'DisplayName','K-path');
    end
%--------plotwhole--------
    for Ei=1:Nbands
        plot(ax,klist,EIGENCAR(Ei,:),'LineWidth',1.0,'Color',color,'DisplayName',num2str(Ei));
        %hold on;
    end
%-------- set label-------
    set(ax,'XTickLabel','')  ;
    set(ax,'FontName',fontname,'FontSize',24,'LineWidth',1,...
    'XTick',kpoints_l,'XTickLabel',...
    kpoints_name);
    ylabel(ax,'E(eV)','FontName',fontname);
%--------  fbug  --------
    titlestring = cell2mat(titlestring);
%--------  title  -------
    title(ax,titlestring);
%--------  save  --------
    %disp(class(titlestring));

    [fig,ax] = save_figure(fig,titlestring+".eps",ax);
%-------- return --------
end
