function [fig,ax]=pbandplot(WEIGHTCAR_struct,EIGENCAR,Ecut,titlestring,cmap,klist_l,kpoints_l,kpoints_name,fontname,fig,ax)
%--------  init  --------
    import vasplib_tool.*
%--------  narg  --------
    if nargin < 10 & nargin > 0
       [fig,ax] = creat_figure();
    end
    if nargin < 9
        fontname="Helvetica";
    end
    if nargin < 8
    end
    if nargin < 7
    end
    if nargin < 6
        [~,klist_l,~,kpoints_l,kpoints_name]=kpathgen3D();
    end
    if nargin < 5
        cmap = jet;
        ncmap = length(cmap);
    end
    if nargin < 4
        dirname = pwd;
        dirname = strsplit(dirname,'/');
        titlestring = dirname(length(dirname));
        titlestring = strrep(titlestring,'_','-');
    end
    if nargin < 3
        Ecut=[-3,3];
    end
    if nargin < 2
        EIGENCAR=EIGENVAL_read();
    end
    if nargin < 1
        % WEIGHTCAR_in_a_file;
        [WEIGHTCAR_struct_cell,Name_list,~] = WEIGHTCAR_gen_dir('PBAND');
        [~,nWEIGHTCAR_struct_cell] = size(WEIGHTCAR_struct_cell);
        for i = 1:nWEIGHTCAR_struct_cell
            titlestring_new = titlestring+"-"+Name_list(i,:);
            disp(titlestring_new );
            WEIGHTCAR_struct = WEIGHTCAR_struct_cell{i};
            % rm the last
            WEIGHTCAR_struct(end) = [];
            [fig_temp,ax_temp]=pbandplot(WEIGHTCAR_struct,EIGENCAR,Ecut,titlestring_new ,cmap,klist_l,kpoints_l,kpoints_name,fontname);
            %--------  fbug  --------
            fig(i) = fig_temp;ax(i) = ax_temp;
        end
        %--------  fbug  --------
        return;
        % diedai ?
    end
%--------  juge  --------

    if nargin >4
        disp('cmap need to be improved');
        if ~isa(cmap,'double')
            color = str2rgb(color);
        end
        ncmap = length(cmap);
    end
%-------- distri --------
    if isa(WEIGHTCAR_struct,'double')
         pbandmode = 'patch'         ;
         WEIGHTCAR = WEIGHTCAR_struct;
    elseif isa(WEIGHTCAR_struct,'cell')
         pbandmode = 'bubble_rough' ;
         WEIGHTCAR_cell = WEIGHTCAR_struct;
         nWEIGHTCAR = length(WEIGHTCAR_cell);
         Name_list(nWEIGHTCAR ,:) = "";
         %--------  bug  --------
         if nWEIGHTCAR > ncmap
             cmap = repmat(cmap,[(round(nWEIGHTCAR/ncmap)+1),1]);
         elseif nWEIGHTCAR < ncmap
             Dcmap = round(ncmap/nWEIGHTCAR);
             for i = 1:nWEIGHTCAR
                 cmap_new(i,:) =cmap((Dcmap-1)*i+i,:);
             end
             cmap = cmap_new;
         end
         for i = 1:nWEIGHTCAR
             Name_list(i,:) = strcat('A',string(i));
         end
    elseif isa(WEIGHTCAR_struct,'struct')
         % it means we can rebuild code for highly user-like plot
         pbandmode = 'bubble_refine';
         nWEIGHTCAR = length(WEIGHTCAR_struct);
         Name_list(nWEIGHTCAR ,:) = "";
         WEIGHTCAR_cell{nWEIGHTCAR} = WEIGHTCAR_struct(nWEIGHTCAR).WEIGHTCAR;
         for i = 1 : nWEIGHTCAR
             WEIGHTCAR_cell{i} =  WEIGHTCAR_struct(i).WEIGHTCAR;
             Name_list(i ,:) = WEIGHTCAR_struct(i).displayname;
             % if color
             % if alpha
         end
         %--------  bug  --------
         if nWEIGHTCAR > ncmap
             cmap = repmat(cmap,[(round(nWEIGHTCAR/ncmap)+1),1]);
         elseif nWEIGHTCAR < ncmap
             Dcmap = round(ncmap/nWEIGHTCAR);
             for i = 1:nWEIGHTCAR
                cmap_new(i,:) =cmap((Dcmap-1)*i+i,:);
             end
             cmap = cmap_new;
         end
    end
%--------  init  --------
    Nbands=size(EIGENCAR,1);
    klist=klist_l;
    xmin=klist_l(1);
    xmax=klist_l(length(klist_l));
    Xcut = [xmin,xmax];
%--------plotwhole--------_
     for Ei=1:Nbands
        plot(ax,klist,EIGENCAR(Ei,:),'LineWidth',1.0,'Color',[0.1 0.1 0.1],'DisplayName',num2str(Ei));
        %hold on;
    end
%--------plotweight--------
    switch pbandmode
        case 'patch'
            [fig,ax] = pband_plot_one_patch(klist,EIGENCAR,WEIGHTCAR,cmap,fig,ax);
        case 'bubble_rough'
            [fig,ax] = pband_plot_set(klist,EIGENCAR,WEIGHTCAR_cell,Name_list,cmap,fig,ax);
        case 'bubble_refine'
            [fig,ax] = pband_plot_set(klist,EIGENCAR,WEIGHTCAR_cell,Name_list,cmap,fig,ax);% waiting
    end
%--------reference -------
    [fig,ax] = set_reference(fig,ax,'band',Xcut,Ecut,kpoints_l,kpoints_name,fontname);
%--------  fbug  --------
    titlestring = cell2mat(titlestring);
%--------  title  -------
    title(ax,titlestring);
%--------  save  --------
    %disp(class(titlestring));

    [fig,ax] = save_figure(fig,titlestring+".eps",ax);
%-------- return --------
end
